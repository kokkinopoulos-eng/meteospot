    }

    if (snow > 20 && wind < 50) {
      parts.add('🏂 Snowboard: Καλές συνθήκες');
    } else {
      parts.add('🏂 Snowboard: Ακατάλληλες συνθήκες');
    }

    if (temp < -5) {
      parts.add('🥶 Πολύ κρύο — ζεστά ρούχα απαραίτητα');
    } else if (temp < 0) {
      parts.add('❄️ Παγετός — προσοχή στους δρόμους');
    }

    parts.add('\n⚠ Ενδεικτική ανάλυση. Συμβουλευτείτε τα χιονοδρομικά κέντρα.');
    return parts.join('\n');
  }
