      case 'good': parts.add('✅ Κολύμπι: Κατάλληλες συνθήκες'); break;
      case 'warning': parts.add('⚠ Κολύμπι: Προσοχή λόγω κυμάτων'); break;
      default: parts.add('❌ Κολύμπι: Ακατάλληλες συνθήκες');
    }
    switch (data.surfRating) {
      case 'good': parts.add('🏄 Σερφ: Κατάλληλες συνθήκες'); break;
      case 'ok': parts.add('🏄 Σερφ: Μέτριες συνθήκες'); break;
      default: parts.add('🏄 Σερφ: Ακατάλληλες συνθήκες');
    }
    if (data.divingRating == 'good') {
      parts.add('🤿 Κατάδυση: Καλές συνθήκες');
    } else {
      parts.add('🤿 Κατάδυση: Ακατάλληλες συνθήκες');
    }
    parts.add('\n⚠ Ενδεικτική ανάλυση. Συμβουλευτείτε τις αρμόδιες αρχές.');
    return parts.join('\n');
  }
